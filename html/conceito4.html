Teste de continuidade

O teste de continuidade é uma forma de verificar se os componentes de um circuito elétrico estão funcionando corretamente, identificando cabos rompidos, curtos ou falhas dentro do sistema utilizando um multímetro.

// C++ code
    //
    int analog = 0;
    
    void setup()
    {
      pinMode(A0, INPUT);
      pinMode(12, OUTPUT);
      Serial.begin(9600);
      pinMode(11, OUTPUT);
      pinMode(10, OUTPUT);
    }
    
    void loop()
    {
      analog = analogRead(A0);
      if (analog >= 300) {
        digitalWrite(12, HIGH);
        Serial.println(analog);
      }
      if (analog >= 600) {
        digitalWrite(12, LOW);
        digitalWrite(11, HIGH);
      }
      if (analog >= 900) {
        digitalWrite(12, LOW);
        digitalWrite(11, LOW);
        digitalWrite(10, HIGH);
      }
      if (analog < 900) {
        digitalWrite(12, LOW);
        digitalWrite(11, HIGH);
        digitalWrite(10, LOW);
      }
      if (analog < 600) {
        digitalWrite(12, HIGH);
        digitalWrite(11, LOW);
        digitalWrite(10, LOW);
      }
      delay(10); // Delay a little bit to improve simulation performance
    }
